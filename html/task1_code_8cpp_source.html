39float medianFilter(float newValue)
40{
41 // Insert the new value into the window
42 window[windowIndex] = newValue;
43 windowIndex = (windowIndex + 1) % WINDOW_SIZE;
44
45 // Copy the window to a temporary array
46 float temp[WINDOW_SIZE];
47 for (int i = 0; i < WINDOW_SIZE; i++)
48 {
49 temp[i] = window[i];
50 }
51
52 // Sort the temporary array
53 for (int i = 0; i < WINDOW_SIZE - 1; i++)
54 {
55 for (int j = i + 1; j < WINDOW_SIZE; j++)
56 {
57 if (temp[i] > temp[j])
58 {
59 float swap = temp[i];
60 temp[i] = temp[j];
61 temp[j] = swap;
62 }
63 }
64 }
65
66 // Return the median value
67 return temp[WINDOW_SIZE / 2];
68}

96static float pressPID(float targetPressure, float currentPressure)
97{
98 unsigned long now = millis();
99 unsigned long timeChange = (now - previousTimeStep);
100 previousTimeStep = now;
101 float error = targetPressure - currentPressure;
102
103 // Calculate the derivative error
104 derivative = (error - prev_error) / timeChange;
105 integral += error * timeChange;
106
107 prev_error = error;
108
109 // The errors are cleared and set to 0 in the IDLE state.
110 outputWithI = KI * integral;
111 outputWithP = KP * error;
112 outputWithD = KD * derivative;
113
114 // Limit the integral term to prevent windup
115 if (outputWithI > 25.0)
116 {
117 integral = 10.0;
118 }
119 else if (outputWithI < -25.0)
120 {
121 integral = -10.0;
122 }
123
124 float output = KP * error + KI * integral + KD * derivative;
125
126 // Constrain output to maximum limits
127 if (output > MAX_OUTPUT)
128 {
129 output = MAX_OUTPUT;
130 }
131 else if (output < -MAX_OUTPUT)
132 {
133 output = -MAX_OUTPUT; // negative speed /direction change
134 }
135
