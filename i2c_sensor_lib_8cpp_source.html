40void RotationTracker_Update(RotationTracker *tracker, uint16_t currentAngle)

135uint16_t readAngle()
136{
137 Wire.beginTransmission(AS5048B_ADDRESS);
138 Wire.write(ANGLE_REGISTER_MSB);
139 if (Wire.endTransmission(false) != 0)
140 {
141 Serial.println("Error writing MSB register");
142 return 0xFFFF; // Return error value
143 }
144
145 Wire.requestFrom(AS5048B_ADDRESS, 1);
146 if (Wire.available() < 1)
147 {
148 Serial.println("Error reading MSB");
149 return 0xFFFF; // Return error value
150 }
151 uint8_t highByte = Wire.read();
152
153 Wire.beginTransmission(AS5048B_ADDRESS);
154 Wire.write(ANGLE_REGISTER_LSB);
155 if (Wire.endTransmission(false) != 0)
156 {
157 Serial.println("Error writing LSB register");
158 return 0xFFFF; // Return error value
159 }
160
161 Wire.requestFrom(AS5048B_ADDRESS, 1);
162 if (Wire.available() < 1)
163 {
164 Serial.println("Error reading LSB");
165 return 0xFFFF; // Return error value
166 }
167 uint8_t lowByte = Wire.read();
168
169 uint16_t angle = ((highByte << 6) | (lowByte & 0x3F)); // 14-bit angle
170
171 return angle;

213void readPressureSensor(float *pressure, float *temperature)
214{
215 uint8_t buffer[4];
216
217 const int maxRetries = 2;
218 bool success = false;
219
220 for (int attempt = 0; attempt < maxRetries; ++attempt)
221 {
222 // Request 4 bytes from the sensor (DF4 format)
223 Wire.beginTransmission(SENSOR_ADDR);
224 if (Wire.endTransmission(false) != 0)
225 {
226 Serial.println("Error: Failed to initiate communication with sensor.");
227 delay(100 * (attempt + 1)); // Exponential backoff
228 continue; // Retry
229 }
230
231 Wire.requestFrom(SENSOR_ADDR, 4);
232
233 // Read the 4 bytes into the buffer
234 if (Wire.available() == 4)
235 {
236 for (int i = 0; i < 4; i++)
237 {
238 buffer[i] = Wire.read();
239 }
240
241 // Extract pressure and temperature from the buffer
242 int16_t raw_pressure = (buffer[0] << 8) | buffer[1];
243 int16_t raw_temperature = (buffer[2] << 8) | (buffer[3] & 0xF0);
244 raw_temperature = raw_temperature >> 4;
245
246 // Convert raw values to human-readable form
247 *pressure = ((float)raw_pressure / PRESSURE_COUNTS) * (PRESSURE_MAX - PRESSURE_MIN) + PRESSURE_OFFSET;
248 *temperature = ((float)raw_temperature / TEMP_COUNTS) * (TEMP_MAX - TEMP_MIN) + TEMP_MIN;
249
250 success = true;
251 break; // Exit the retry loop on success
252 }
253 else
254 {
255 Serial.println("Error: Did not receive 4 bytes of data from sensor.");
256 delay(100 * (attempt + 1)); // Exponential backoff
257 }
258 }
259
260 // End the I2C transmission
261 Wire.endTransmission();
262
263 if (!success)
264 {
265 Serial.println("Error: Failed to read sensor data after multiple attempts.");
266 }
267}
