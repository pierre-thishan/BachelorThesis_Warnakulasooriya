36static float calculateTempPID(float temperature)
37{
38 unsigned long now = millis();
39 unsigned long timeChange = (now - tempPreviousTimeStep);
40
41 float error = setpoint - temperature;
42 float derivative = (error - tempPrev_error) / timeChange;
43
44 // Integral with anti-windup clamping
45 tempIntegral += error * timeChange;
46 if (tempIntegral > INTEGRAL_WINDUP_LIMIT)
47 tempIntegral = INTEGRAL_WINDUP_LIMIT;

49 tempIntegral = -INTEGRAL_WINDUP_LIMIT;

51 float output = TKp * error + TKi * tempIntegral + TKd * derivative;
52 tempPrev_error = error;
53 tempPreviousTimeStep = now;
