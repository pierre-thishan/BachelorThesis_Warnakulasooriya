285void processCommand(String command)
286{
287 // Split the command into tokens
288 char *cmd = strtok(&command[0u], " ");
289 while (cmd != NULL)
290 {
291 String token = String(cmd);
292 char *valueStr = strtok(NULL, " "); // Get the next token (expected value)
293
294 if (token == "press")
295 {
296 if (valueStr != NULL)
297 {
298 float value = atof(valueStr); // Convert value to float
299 if (value >= 0 && value <= 12000)
300 {
301 targetPressure = value; // Assign value to targetPressure
302
303 if (myservo.read() != 180)
304 {
305 servoMove(180);
306 }
307
308 GPStopFlag = false;
309 bKeepPressingFlag = false;
310 bPlaceHomeFlag = false;
311
312 setCurrentRoutine(PRESS); // State transition to PRESS
313 }
314 else
315 {
316 Serial.println("Error: 'press' value out of range (0-12000).");
317 }
318 }
319 else
320 {
321 Serial.println("Error: Invalid or missing value for 'press' command.");
322 }
323 break;
324 }
325 else if (token == "GPStop")
326 {
327 GPStopFlag = true;
328
329 bKeepPressingFlag = false;
330 bPlaceHomeFlag = false;
331
332 setCurrentRoutine(STOP); // State transition to STOP
333 break;
334 }
335
336 else if (token == "P" || token == "D" || token == "I" ||
337 token == "TmpP" || token == "TmpD" || token == "TmpI")
338 {
339 if (valueStr != NULL)
340 {
341 float value = atof(valueStr);
342 if (value >= 0 && value <= 12000)
343 {
344 if (token == "P")
345 KP = value;
346 if (token == "D")
347 KD = value;
348 if (token == "I")
349 KI = value;
350 if (token == "TmpP")
351 TKp = value;
352 if (token == "TmpD")
353 TKd = value;
354 if (token == "TmpI")
355 TKi = value;
356 }
357 else
358 {
359 Serial.println("Error: PID value out of range (0-12000).");
360 }
361 }
362 else
363 {
364 Serial.println("Error: Invalid or missing value for PID command.");
365 }
366 }
367
368 else if (token == "home")
369 {
370 GPStopFlag = false;
371 bKeepPressingFlag = false;
372 bPlaceHomeFlag = false;
373
374 setCurrentRoutine(HOMING); // State transition to HOMING
375 break;
376 }
377 else if (token == "keepPressing")
378 {
379 GPStopFlag = false;
380 bPlaceHomeFlag = false;
381 bKeepPressingFlag = !bKeepPressingFlag; // toggle the flag
382 if (bKeepPressingFlag)
383 {
384 setCurrentRoutine(KEEP_PRESSING);
385 }
386 else
387 {
388 setCurrentRoutine(IDLE);
389 }
390 break;
391 }
392 else if (token == "temperature")
393 {
394 if (valueStr != NULL)
395 {
396 float value = atof(valueStr);
397 if (value >= 0 && value <= 180)
398 {
399 setpoint = value; // Assign value to setpoint
400 }
401 else
402 {
403 Serial.println("Error: 'temperature' value out of range (0-180).");
404 }
405 }
406 else
407 {
408 Serial.println("Error: Invalid or missing value for 'temperature' command.");
409 }
410 break;
411 }
412 else if (token == "servo")
413 {
414 if (valueStr != NULL)
415 {
416 int angle = atoi(valueStr); // Convert value to int
417 if (angle == 0 || angle == 180)
418 {
419 servoMove(angle); // Move servo to specified angle
420 }
421 else
422 {
423 Serial.println("Error: Servo angle must be 0 or 180.");
424 }
425 }
426 else
427 {
428 Serial.println("Error: Invalid or missing value for 'servo' command.");
429 }
430 break;
431 }
432 else if (token == "pick")
433 {
434 if (myservo.read() != 180)
435 {
436 servoMove(180);
437 }
438 digitalWrite(pump_pwm_pin, HIGH);
439 setTargetPressure(2500.0);
440
441 GPStopFlag = false;
442 bKeepPressingFlag = false;
443 bPlaceHomeFlag = false;
444
445 vTaskDelay(pdMS_TO_TICKS(100));
446 setCurrentRoutine(PICKING);
447 break;
448 }
449 else if (token == "place")
450 {
451 GPStopFlag = false;
452 bKeepPressingFlag = false;
453 bPlaceHomeFlag = false;
454 setTargetPressure(2300.0);
455 setCurrentRoutine(PLACING);
456 break;
457 }
458 else if (token == "pump")
459 {
460 if (valueStr != NULL)
461 {
462 int on_off = atoi(valueStr); // Convert value to int
463 if (on_off == 0 || on_off == 1)
464 {
465 digitalWrite(pump_pwm_pin, on_off); // Set pump state
466 }
467 else
468 {
469 Serial.println("Error: Pump value must be 0 or 1.");
470 }
471 }
472 else
473 {
474 Serial.println("Error: Invalid or missing value for 'pump' command.");
475 }
476 break;
477 }
478 else if (token == "voltage")
479 {
480 if (valueStr != NULL)
481 {
482 float value = atof(valueStr); // Convert value to float
483 if (value >= 0 && value <= 8)
484 {
485 motor.voltage_limit = value; // Assign value to voltage_limit
486 }
487 else
488 {
489 Serial.println("Error: 'voltage' value out of range (0-8).");
490 }
491 }
492 else
493 {
494 Serial.println("Error: Invalid or missing value for 'voltage' command.");
495 }
496 break;
497 }
498 else if (token == "move")
499 {
500 if (valueStr != NULL)
501 {
502 float value = atof(valueStr); // Convert value to float
503 if (value >= traverseMinBound && value <= traverseMaxBound)
504 {
505 traverseTargetPosition = value; // Assign value to traverseTargetPosition
506 GPStopFlag = false;
507 bKeepPressingFlag = false;
508 bPlaceHomeFlag = false;
509 setCurrentRoutine(TRAVERSE);
510 }
511 else
512 {
513 Serial.println("Error: 'move' position out of bounds.");
514 }
515 }
516 else
517 {
518 Serial.println("Error: Invalid or missing value for 'move' command.");
519 }
520 break;
521 }
522 else
523 {
524 StaticJsonDocument<256> doc;
525 doc["status"] = "error";
526 serializeJson(doc, Serial);
527 Serial.println();
528 }
529 cmd = strtok(NULL, " "); // Get the next token
530 }
